\begin{frame}
\ctitle{OBJETIVOS}
\vspace{-1mm}
{\footnotesize\alert{Objetivo General}}\\[1mm]
\footnotesize\justifying
Comparar los resultados estructurales de un galpón industrial de acero
aplicando las versiones actualizadas de \alert{NCh2369} (sismo) y
\alert{NCh432} (viento) respecto de sus versiones anteriores.

\vspace{2.5mm}
{\footnotesize\alert{Objetivos Específicos}}\\[1mm]
\begin{itemize}\footnotesize\setlength{\itemsep}{1mm}\setlength{\parsep}{0pt}
  \item Validación constructiva estructural del galpón existente
        (levantamiento).
  \item Verificación del diseño normativo del galpón existente.
  \item Evaluación variación sísmica: \alert{NCh2369}.
  \item Evaluación variación de viento: \alert{NCh432}.
  \item Comparación: cortes basales, desplazamientos y factores de
        utilización (miembros principales).
\end{itemize}
\end{frame}

\section{CONCLUSIONES Y COMENTARIOS}

%------------------------------------------------------------------
% 17. CONCLUSIONES Y COMENTARIOS
%------------------------------------------------------------------
\begin{frame}
\ctitle{CONCLUSIONES Y COMENTARIOS}
\begin{itemize}\footnotesize\setlength{\itemsep}{2.5mm}
  \item Se espera que \alert{NCh432:2025} entregue \alert{menores} cargas de
        viento que \alert{NCh432Of.1971}, dado el mayor refinamiento
        metodológico de la norma actualizada.
  \item Se espera que \alert{NCh2369:2025} entregue \alert{mayores} cortes
        basales respecto de \alert{NCh2369Of.2003}, según el tipo de suelo
        y zona sísmica.
  \item La investigación cuantificará la variación en \alert{cortes basales},
        \alert{desplazamientos} y \alert{factores de utilización} de perfiles
        y conexiones.
  \item Se espera identificar los miembros y conexiones que requieren
        adecuación bajo la normativa vigente.
\end{itemize}
\end{frame}
